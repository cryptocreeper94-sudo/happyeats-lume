// server/ecosystemClient.ts
export class TrustLayerHubClient {
  private apiKey = process.env.TRUSTLAYER_HUB_API_KEY || '';
  private apiSecret = process.env.TRUSTLAYER_HUB_API_SECRET || '';
  private hubUrl = 'https://orbitstaffing.io';
  
  getHeaders() {
    return {
      'Content-Type': 'application/json',
      'X-API-Key': this.apiKey,
      'X-API-Secret': this.apiSecret,
      'X-App-Name': 'Trust Layer Driver Connect'
    };
  }
  
  async checkConnection(): Promise<boolean> {
    const res = await fetch(`${this.hubUrl}/api/ecosystem/status`, {
      headers: this.getHeaders()
    });
    return res.ok;
  }
  
  async logActivity(action: string, category: string, details: any): Promise<void> {
    await fetch(`${this.hubUrl}/api/ecosystem/logs`, {
      method: 'POST',
      headers: this.getHeaders(),
      body: JSON.stringify({ action, category, details, timestamp: new Date().toISOString() })
    });
  }
}

// server/solanaStamp.ts
import { Connection, Keypair, Transaction, SystemProgram, PublicKey, sendAndConfirmTransaction } from '@solana/web3.js';
import crypto from 'crypto';
export class TrustLayerBlockchain {
  private connection: Connection;
  private wallet: Keypair | null = null;
  
  constructor() {
    const heliusKey = process.env.HELIUS_API_KEY;
    this.connection = new Connection(`https://mainnet.helius-rpc.com/?api-key=${heliusKey}`);
    
    const privateKey = process.env.SOLANA_WALLET_PRIVATE_KEY;
    if (privateKey) {
      this.wallet = Keypair.fromSecretKey(Buffer.from(JSON.parse(privateKey)));
    }
  }
  
  generateHash(data: string): string {
    return crypto.createHash('sha256').update(data).digest('hex');
  }
  
  async stampVersion(version: string, appName: string): Promise<{ hash: string; signature: string | null }> {
    const data = JSON.stringify({ version, appName, timestamp: new Date().toISOString() });
    const hash = this.generateHash(data);
    
    if (!this.wallet) return { hash, signature: null };
    
    try {
      const transaction = new Transaction().add(
        SystemProgram.transfer({
          fromPubkey: this.wallet.publicKey,
          toPubkey: this.wallet.publicKey,
          lamports: 0
        })
      );
      
      // Add memo instruction
      const memoProgram = new PublicKey('MemoSq4gqABAXKb96qnH8TysNcWxMyWCqXgDLGmfcHr');
      transaction.add({
        keys: [],
        programId: memoProgram,
        data: Buffer.from(`TrustLayer:${hash}`)
      });
      
      const signature = await sendAndConfirmTransaction(this.connection, transaction, [this.wallet]);
      return { hash, signature };
    } catch (error) {
      console.error('Blockchain stamp failed:', error);
      return { hash, signature: null };
    }
  }
}
